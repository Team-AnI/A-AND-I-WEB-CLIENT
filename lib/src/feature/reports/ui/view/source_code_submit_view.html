# A&I 터미널 계산기 과제 제출 def calculate(): print("계산기 프로그램을 시작합니다.") while True: try: num1 = float(input("첫 번째 숫자: ")) num2 = float(input("두 번째 숫자: ")) op = input("연산자 (+, -, *, /): ") if op == '+': result = num1 + num2 elif op == '-': result = num1 - num2 elif op == '*': result = num1 * num2 elif op == '/': result = num1 / num2 else: print("잘못된 연산자입니다.") continue print(f"결과: {result}") except ZeroDivisionError: print("Error: 0으로 나눌 수 없습니다.") except ValueError: print("Error: 올바른 숫자를 입력하세요.") if __name__ == "__main__": calculate()
